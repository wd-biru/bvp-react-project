import * as React from "react";
import styled from "styled-components";
import theme from "../theming/Theming";
 
const setButtonStyle = props => {
  switch (props.buttonType) {
    case "primary":
      return `background-color: ${theme.colors.primary};
      border: 1px solid ${theme.colors.primary};
      color: ${theme.colors.textPrimary};
    `;
    case "secondary":
      return `background-color: ${theme.colors.secondary};
    border: 1px solid ${theme.colors.secondary};
    color: ${theme.colors.textSecondary};
  `;
    default:
      return `background-color: ${theme.colors.default};
      border: 1px solid ${theme.colors.defaultBorder};
      color: ${theme.colors.primary};
  `;
  }
};
 
const StyledButton = styled.button`
  border-radius: ${theme.radius.radius};
  padding: ${theme.frm.elemPad};
  ${props => setButtonStyle(props)};
`;
 
class ButtonUi extends React.PureComponent {
  render() {
    const { buttonType, children, ...props } = this.props;
    return (
      <StyledButton
        type="button"
        onClick={this.props.click}
        buttonType={buttonType}
        {...props}
      >
        {children}
      </StyledButton>
    );
  }
}
 
export default ButtonUi;
