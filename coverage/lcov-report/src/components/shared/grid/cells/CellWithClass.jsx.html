import * as React from 'react';
import PropTypes from 'prop-types';
import styled from 'styled-components';
import theme from '../../theming/Theming'
 
const Td = styled.td`
    text-align: left;
    .open:before,
    .in-progress:before {
        content: "";
        width: 8px;
        height: 8px;
        display: inline-block;
        margin-right: 10px;
        border-radius: 10px;
    }
    .open {
        &:before {            
            background-color: ${theme.colors.secondary};
        } 
    }
    .in-progress {
        &:before {
            background-color: ${theme.colors.primary};
        } 
    }
`
 
export default class CellWithClass extends React.PureComponent {
 
    getClassName = (data) => {
        const className = (data.replace(' ', '-')).toLowerCase();
        return className;
    }
    render() {
        return (
            <Td>
                <span className={this.getClassName(this.props.data)}>
                    {this.props.data ? this.props.data : ''}
                </span>
            </Td>
        );
    }
}
