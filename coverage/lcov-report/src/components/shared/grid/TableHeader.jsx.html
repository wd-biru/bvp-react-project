import * as React from 'react';
import PropTypes from 'prop-types';
import { SortingDirection } from './TableConstant';
import styled from 'styled-components';
 
const TableHeaderCSS = styled.thead`
  
  th:first-child{border-top-left-radius: 5px;}
  th:last-child{border-top-right-radius: 5px;}
 
  th, td{
    background:#efefef;
    border-bottom:1px solid #d1d1d1;
    padding:15px 1%;
    text-align:center;
    font-weight:600;
    &.left{text-align:left;}
    &.right{text-align:right;}
  }
  .sort-Wrapper {
    position: relative;
    display: inline-block;
    margin-left: 10px;
 
    &.active.asc .asc {
      border-bottom-color: red;
    }
  
    &.active.desc .desc {
      border-top-color: red;
    }
  }
  .sorting {
    width: 0; 
    height: 0;
    position: absolute;
 
    &.asc {
      border-left: 5px solid transparent;
      border-right: 5px solid transparent;  
      border-bottom: 6px solid #666666;
      top: -10px
    }
 
    &.desc {
      border-left: 5px solid transparent;
      border-right: 5px solid transparent;  
      border-top: 6px solid #666666;
      top: -2px;
    }
  }
 
`;
 
export default class TableHeader extends React.PureComponent {
 
  constructor(props) {
    super(props);
    this.state = {
      sort: {
        columnName: '',
        direction: ''
      }
    };
    this.applyGuidedSort = this.applyGuidedSort.bind(this);
    this.toggleSortingIcon = this.toggleSortingIcon.bind(this);
  }
 
  applyGuidedSort(columnName) {
    if (this.state.sort.columnName !== columnName) {
      this.setState({sort: {columnName, direction: '' }});
    }
    const direction = (this.state.sort.direction === 'ASC') ? 'DESC' : 'ASC';
    this.setState({sort: {columnName, direction }});
    this.props.onSortTable(this.state.sort);
  }
 
  toggleSortingIcon(columnName) {
    let sortingIconClass = '';
    if(this.state.sort.direction === 'ASC' && this.state.sort.columnName === columnName ) {
      return sortingIconClass = 'active asc';
    } else  if(this.state.sort.direction === 'DESC' && this.state.sort.columnName === columnName) {
      return sortingIconClass = 'active desc';
    }
    return sortingIconClass;
 
  }
 
 
  render() {
 
    return (
      <TableHeaderCSS>
        <tr>{this.props.columns.map((column) => {
          return (
            <th key={`${column.bindingProperty}`} onClick={() => this.applyGuidedSort(column.bindProperty)}>
              {column.displayName}
              {column.sorting ? <span className={`sort-Wrapper ${this.toggleSortingIcon(column.bindProperty)}`}>
                <span className="sorting asc"></span>
                <span className="sorting desc"></span></span> : ''}
            </th>
          );
        })}
        </tr>
      </TableHeaderCSS>
    );
  }
}
 
TableHeader.propTypes = {
  columns: PropTypes.object,
  onSortTable: PropTypes.func
};
 
TableHeader.defaultProps = {
  columns: {}
};
