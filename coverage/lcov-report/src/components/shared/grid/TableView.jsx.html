import * as React from 'react';
import TextCell from './cells/TextCell';
import { tableCellType } from './TableConstant';
import styled from 'styled-components';
import CellWithClass from './cells/CellWithClass';
 
const TableBodyCss = styled.tbody`
  td{
    background:none;
    border:0;
    font-size:14px;
    &.red{color: #ff3333;}
    &.blue{color:#0066cc;}
  }
 
  .status-icon{
    width:12px;
    height:12px;
    border-radius:6px;
    display:inline-block;
    margin-right:10px;
    &.open{background:#ff3333;}
    &.in-progress{background:#00cc00;}
  }
 
`;
 
export default class TableView extends React.PureComponent {
 
  renderCell(column, rowData) {
    switch (column.type) {
      case tableCellType.text: {
        return <TextCell data={rowData[column.bindProperty]}
         column={column}
          />;
      }
      case tableCellType.classCell: {
        return <CellWithClass data={rowData[column.bindProperty]}
         column={column}
          />;
      }
      default:
        break;
    }
  }
  render() {
    return (
      <TableBodyCss>
 
        {
          this.props.data.map(rowData => {
            return (
              <tr>
               { this.props.columns.map(column => {
                  return (
                  this.renderCell(column, rowData)
                 );
 
                }) }
                </tr>
                );
                
            })
        }
      </TableBodyCss>
    );
  }
}
