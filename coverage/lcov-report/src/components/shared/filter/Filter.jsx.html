import React from "react";
import styled from "styled-components";
import filtericon from "../../../assets/img/filter-icon.png";
import Modal from "../modal/ModalWrapper";
import FilterContent from "./FilterContent";
 
const FilterGrid = styled.button`
  display: flex;
  align-items: center;
  justify-content: space-between;
  border: 0;
  padding: 10px 0;
  background-color: transparent;
 
  .text {
    margin-right: 10px;
  }
`;
 
const FilterWrapper = styled.div`
  position: relative;
`;
 
class Filter extends React.PureComponent {
  constructor(props) {
    super(props);
    this.modalData = {
      title: "Filter"
    };
    this.state = {
      open: false
    };
    this.onOpenModal = this.onOpenModal.bind(this);
  }
 
  /*
  //Grid filter show hide
  */
  onOpenModal = () => {
    this.setState({
      open: !this.state.open
    });
  };
 
  hideModal = () => {
    this.setState({
      open: false
    });
  }
  
 
  render() {
    return (
      <FilterWrapper>
        <FilterGrid onClick={this.onOpenModal}>
          <span className="text">Filter by</span>{" "}
          <img src={filtericon} alt="" />
        </FilterGrid>
        {this.state.open ? 
          <Modal titleData={this.modalData.title } handleClose={this.hideModal}>
            <FilterContent />
          </Modal> : null 
        }
      </FilterWrapper>
    );
  }
}
 
export default Filter;
