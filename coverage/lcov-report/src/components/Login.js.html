import React from 'react';
import PropTypes from 'prop-types';
import { withStyles } from '@material-ui/core/styles';
import Button from '@material-ui/core/Button';
import TextField from '@material-ui/core/TextField';
import { connectToStore } from '../external/kohls-platform-lib/connect';
import { doLogin } from '../actionCreators/auth';
 
class Login extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      login: {
        userName: '',
        password: ''
      }
    };
  }
 
  handleChange = prop => event => {
    this.setState({
      login: {
        ...this.state.login,
        [prop]: event.target.value
      }
    });
  };
 
  handleSubmit = e => this.props.doLogin(this.state.login);
 
  render() {
    const { login } = this.state;
    return (
      <div className="login-container">
        <h1>Login</h1>
        <div>
          <TextField
            label="UserName"
            autoComplete="off"
            defaultValue={login.userName}
            onChange={this.handleChange('userName')}
          />
          <TextField
            type="password"
            label="Password"
            defaultValue={login.password}
            onChange={this.handleChange('password')}
          />
        </div>
        <Button onClick={this.handleSubmit}>Login</Button>
      </div>
 
    );
  }
}
 
Login.propTypes = {
  classes: PropTypes.object.isRequired,
};
 
const styles = theme => ({
  container: {
    display: 'flex',
    flexWrap: 'wrap',
  },
  input: {
    margin: theme.spacing.unit,
  },
});
 
 
export default withStyles(styles)(connectToStore(Login, null, { doLogin }));
